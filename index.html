MethodEndpointDescription
GET api/anime-ids.json 3,255+ MAL IDs — ids[], count, updated, endpoints_run
GET api/sites.json Live streaming sites — name, url, tag, top?, categories{}
GET api/config.json Featured IDs, endpoint URLs, AnimeWeb URL, version
USAGE EXAMPLE
const ANIMEDB = 'https://ccguvycu.github.io/animedb-site';

// Anime catalog
const { ids, count } = await fetch(`${ANIMEDB}/api/anime-ids.json`).then(r => r.json());
// ids -> [1, 19, 20, ...3255 total]

// Live streaming sites
const { sites } = await fetch(`${ANIMEDB}/api/sites.json`).then(r => r.json());
// sites -> [{ name:'HiAnime', url:'...', tag:'sub', top:true }, ...]

// Filter by tag
const subSites = sites.filter(s => s.tag === 'sub');
